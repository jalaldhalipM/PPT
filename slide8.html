React Events

camelCase Syntax

React events are named using camelCase, not lowercase (e.g., onClick vs onclick).

Pass a function definition as the event handler, not a string.

e.preventDefault()

Required to stop default browser behaviors (like page reloads on form submit).

function handleSubmit(e) { e.preventDefault(); }

import { useState } from 'react';

export default function Form() {
  const [name, setName] = useState('');

  function handleSubmit(e) {
    e.preventDefault(); // Stop reload
    alert(`Submitting: ${name}`);
  }

  return (
    <form onSubmit={handleSubmit}>
      <label>Name:</label>
      
      <input 
        value={name}
        onChange={e => setName(e.target.value)}
        placeholder="Enter name"
      />

      <button type="submit">Send</button>
    </form>
  );
}
